%========== Auxiliary definitions specific to this document  ==========%

%----- Miscellaneous -----%
% Force blank page
\newcommand{\blankpage}{\newpage\null\thispagestyle{empty}}

%----- Incorporating external documents -----%
% Publication insertion
\newcommand{\currentPublicationName}{<Publication Title>}
\newcommand{\InsertPublication}[1]{
    \renewcommand{\currentPublicationName}{#1}
    \stepcounter{publication}
    \includepdf[
        pages=-,
        addtotoc={%
            1, % page number
            section, % section
            1, % level
            Publication~\arabic{publication}, %heading
            Appendix-Publication\arabic{publication} % label
            },
        pagecommand={\thispagestyle{publication}},
        width=1.15\textwidth
    ]{Content/BackMatter/AppendixFiles/Publication\arabic{publication}.pdf}
}

% Function for inserting background images
\newcommand{\InsertChapterWallpaper}[1]{
    \cleartoevenpage
    \thispagestyle{empty}
    \ThisCenterWallPaper{1.0}{#1}
}
