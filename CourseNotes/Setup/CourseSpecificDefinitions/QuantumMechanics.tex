% Bra notation
%\newcommand{\bra}[1]{\left\langle{#1}\right|}
\newcommand{\bra}[1]{\left\langle#1\right\rvert}
\newcommand{\smallbra}[1]{\langle#1\vert}

% Ket notation
\newcommand{\ket}[1]{\left\lvert#1\right\rangle}
\newcommand{\smallket}[1]{\vert#1\rangle}

% '< | >' braket notation
%\newcommand{\braket}[2]{%
%	\left\langle{#1}\,\middle|\,{#2}\right\rangle}
\newcommand{\braket}[2]{%
\left\langle#1\vphantom{#2}\right|%
\hspace{-0.9mm}\left.#2\vphantom{#1}%
\right\rangle}

\newcommand{\smallbraket}[2]{%
\langle#1\vphantom{#2}\vert%
\hspace{-0.3mm}#2\vphantom{#1}%
\rangle}

\newcommand{\ketbra}[2]{%
\left\vert#1\vphantom{#2}\right\rangle\hspace{-0.8mm}%
\left\langle#2\vphantom{#1}\right\vert}

\newcommand{\smallketbra}[2]{%
\vert#1\rangle\hspace{-0.2mm}%
\langle#2\vphantom{#1}\vert}
